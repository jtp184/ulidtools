def format_router(fmt)
  case fmt.downcase
  when 'ulid'
    [
      ->(u) { u.length == 26 },
      ->(u) { %w[I L O U].none? { |l| u.include?(l) } }
    ]
  when 'uuid'
    [
      ->(u) { u.match? /[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}/i }
    ]
  when 'bytes'
    [
      ->(u) { !u.ascii_only? }
    ]
  else
    raise ArgumentError, "Undefined Format #{fmt}"
  end
end

def message_router(msg)
  case msg.downcase
  when 'generate'
    :generate
  when 'string'
    :to_s
  when 'bytestring'
    :raw
  when 'uuid'
    :to_uuid
  else
    raise ArgumentError, "Undefined Message #{msg}"
  end
end

def object_router(obj)
  case obj.downcase
  when 'generator'
    ULIDTools
  when 'ulid'
    @ulid
  else
    raise ArgumentError, "Undefined Object #{obj}"
  end
end

def sane_time(time)
  (0..5).cover?(time.to_i - Time.now.to_i)
end
