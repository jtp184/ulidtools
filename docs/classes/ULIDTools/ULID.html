def time
  @time unless @time.nil?
  (ms,) = "\x0\x0#{raw[0..6]}".unpack('Q>')
  @time = Time.at(ms / 1000.0).utc
end

# File lib/ulidtools/ulid.rb, line 67
def generate_time_bytes(time)
  millis = time.to_f
               .*(1000)
               .to_i

  [millis].pack('Q>')[2..-1]
end
ulid_bitmath ()

# File lib/ulidtools/ulid.rb, line 56
def ulid_bitmath
  (x, y) = raw.unpack('Q>Q>')
  (x << 64) | y
end
